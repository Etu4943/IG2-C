function citeNFois (chaine, nbr) {
  nbr = Number(nbr);
  if (!isFinite(nbr) || nbr < 0) {
    sortir("Le deuxième argument doit être un nombre positif");
    return;
  } 
  for (let i = 1; i <= nbr; i++) {
    sortir(`${i}) ${chaine}`);
  }
}
