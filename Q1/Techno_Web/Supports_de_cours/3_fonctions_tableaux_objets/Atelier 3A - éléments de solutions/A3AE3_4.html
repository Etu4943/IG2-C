Version utilisant pop et push


let pile = [];
let dossier = prompt("Nom du dossier ?");
while (dossier != null) {
  if (dossier !== "") {     
    document.write(`
Nouveau dossier : ${dossier} - actuellement :
                    ${pile.push(dossier)} dossier(s) dans la pile
`);
  } else {
    if (pile.length == 0) {
      document.write(`
Plus de dossier dans la pile ! 
`);
    } else {
      document.write(`
Dossier en cours : ${pile.pop(dossier)} - actuellement :
                      ${pile.length} dossier(s) dans la pile
`);
    }
  }
  dossier = prompt("Nom du dossier ?");
}
